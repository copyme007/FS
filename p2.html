import turtle
import math

screen=turtle.Screen()
screen.bgcolor("black")

t=turtle.Turtle()
t.speed(10)
t.pensize(2)
def draw_rect(x,y,w,h,c):
    t.penup()
    t.goto(x,y)
    t.pendown()
    t.color(c)
    for _ in range(2):
        t.forward(w)
        t.left(90)
        t.forward(h)
        t.left(90)

def draw_cir(x,y,r,c):
    t.penup()
    t.goto(x,y)
    t.pendown()
    t.color(c)
    t.circle(r)

def translate(x,y,dx,dy):
    t.penup()
    t.goto(x+dx,y+dy)
    t.pendown()

def rotate(x,y,angle):
    t.penup()
    t.goto(x,y)
    t.setheading(angle)
    t.pendown()

def scale(x,y,sx,sy):
    t.penup()
    t.goto(x*sx,y*sy)
    t.pendown()

draw_rect(-200,0,100,50,"blue")
translate(-200,0,200,0)
draw_rect(0,0,100,50,"green")
scale(0,0,2,2)
draw_rect(-100,100,200,100,"green")
rotate(0,0,45)
draw_rect(0,0,100,50,"green")

draw_cir(100,100,50,"red")
translate(100,100,200,0)
draw_cir(300,100,50,"black")
rotate(300,100,45)
draw_cir(300,100,50,"white")
scale(0,0,2,2)
draw_cir(600,200,50,"black")

turtle.done()
